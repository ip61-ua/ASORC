Add-WindowsCapability -Online -Name OpenSSH.Server~~~~0.0.1.0

Start-Service sshd

Set-Service -Name sshd -StartupType 'Automatic'

Restart-Service sshd

# Desde el host
scp /home/ibai/.ssh/id_rsa.pub Administrador@192.168.25.12:id_rsa.pub.temp

ssh Administrador@192.168.25.12

# Poner PubkeyAuthentication yes
winget install GNU.Nano
nano C:\ProgramData\ssh\sshd_config

# Carpeta .ssh existe?
if (-not (Test-Path -Path $env:USERPROFILE\.ssh)) {
	mkdir $env:USERPROFILE\.ssh
}

# Agregar clave al archivo authorized_keys
Get-Content .\id_rsa.pub.temp | Out-File -FilePath $env:USERPROFILE\.ssh\authorized_keys -Encoding ascii -Append

# Borra el archivo temporal
Remove-Item .\id_rsa.pub.temp
